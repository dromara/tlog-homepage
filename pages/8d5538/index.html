Log4j2жЎҶжһ¶йҖӮй…ҚеҷЁ

иӯҰе‘Ҡ

з”ұдәҺlog4j2жҳҜжҸ’д»¶зҡ„жңәеҲ¶пјҢдёәдәҶйҒҝе…Қlog4j2жҸҗеүҚеҠ иҪҪеҲ°иҮӘе·ұзҡ„жҸ’д»¶пјҢиҜ·жҠҠlog4j2й…ҚзҪ®ж–Үд»¶зҡ„patternдёӯзҡ„m/msg/messageж”№жҲҗtm/tmsg/tmessage

еҰӮжһңдҪ дёҚжғіж”№иҝҷдёӘпјҢиҝҳжңүдёҖдёӘеҠһжі•пјҢе°ұжҳҜжҠҠtlogзҡ„дҫқиө–еҢ…еңЁpomж–Үд»¶йҮҢжҸҗеҲ°log4j2дҫқиө–еҢ…зҡ„еүҚйқўгҖӮ

# еҗҢжӯҘж—Ҙеҝ—

жҸҗзӨә

log4J2з”ұдәҺжҳҜйҖҡиҝҮжҸ’д»¶еҪўејҸе®һзҺ°зҡ„пјҢlog4J2жңүиҮӘеҠЁжү«жҸҸжҸ’д»¶зҡ„еҠҹиғҪгҖӮжүҖд»ҘеңЁеҗҢжӯҘж—Ҙеҝ—жғ…еҶөдёӢж— йңҖеҜ№й…ҚзҪ®ж–Үд»¶еҒҡд»»дҪ•жӣҙж”№е°ұиғҪз”ҹж•ҲгҖӮ

дёӢйқўд»…д»…з»ҷеҮәlog4j2еҗҢжӯҘж—Ҙеҝ—й…ҚзҪ®ж–Үд»¶зҡ„зӨәдҫӢпјҢе…¶е®һе№¶дёҚйңҖиҰҒж”№еҠЁпјҡ

<?xml version="1.0" encoding="UTF-8"?>
<configuration status="WARN" monitorInterval="30">
    <!--е…Ҳе®ҡд№үжүҖжңүзҡ„appender-->
    <appenders>
        <!--жҺ§еҲ¶еҸ°ж—Ҙеҝ—-->
        <console name="Console" target="SYSTEM_OUT">
            <!--иҫ“еҮәж—Ҙеҝ—зҡ„ж јејҸ-->
            <PatternLayout pattern="%date{HH:mm:ss.SSS}  %-5level [%thread] %logger{36} - %tmsg%n"/>
        </console>

        <!--ж–Үд»¶ж—Ҙеҝ—-->
        <!--еҗҢжӯҘж—Ҙеҝ—-->
        <RollingFile name="RollingFileInfo" fileName="./logs/log4j2-sync-rolling.log"
                     filePattern="${sys:user.home}/logs/$${date:yyyy-MM}/info-%d{yyyy-MM-dd}-%i.log">
            <!--жҺ§еҲ¶еҸ°еҸӘиҫ“еҮәlevelеҸҠд»ҘдёҠзә§еҲ«зҡ„дҝЎжҒҜпјҲonMatchпјүпјҢе…¶д»–зҡ„зӣҙжҺҘжӢ’з»қпјҲonMismatchпјү-->
            <ThresholdFilter level="info" onMatch="ACCEPT" onMismatch="DENY"/>
            <PatternLayout pattern="%date{HH:mm:ss.SSS}  %-5level [%thread] %logger{36} - %tmsg%n"/>
            <Policies>
                <TimeBasedTriggeringPolicy/>
                <SizeBasedTriggeringPolicy size="100 MB"/>
            </Policies>
        </RollingFile>

        <!--еҗҢжӯҘж—Ҙеҝ—-->
        <File name="SyncFile" fileName="./logs/log4j2-sync.log" append="false" >
            <PatternLayout pattern="%date{HH:mm:ss.SSS}  %-5level [%thread] %logger{36} - %tmsg%n"/>
        </File>

    </appenders>
    <!--з„¶еҗҺе®ҡд№үloggerпјҢеҸӘжңүе®ҡд№үдәҶloggerе№¶еј•е…Ҙзҡ„appenderпјҢappenderжүҚдјҡз”ҹж•Ҳ-->
    <loggers>
        <root level="INFO">
            <appender-ref ref="Console"/>
            <appender-ref ref="SyncFile"/>
            <appender-ref ref="RollingFileInfo"/>
        </root>
    </loggers>
</configuration>

# ејӮжӯҘж—Ҙеҝ—

жҸҗзӨә

log4J2з”ұдәҺжҳҜйҖҡиҝҮжҸ’д»¶еҪўејҸе®һзҺ°зҡ„пјҢlog4J2жңүиҮӘеҠЁжү«жҸҸжҸ’д»¶зҡ„еҠҹиғҪгҖӮеңЁејӮжӯҘж—Ҙеҝ—жғ…еҶөдёӢд№ҹж— йңҖеҜ№й…ҚзҪ®ж–Үд»¶еҒҡд»»дҪ•жӣҙж”№е°ұиғҪз”ҹж•ҲгҖӮ

дҪҶжҳҜејӮжӯҘж—Ҙеҝ—йҮҢпјҢз”ұдәҺlog4j2жҳҜеҸ‘йҖҒеҲ°йҳҹеҲ—йҮҢпјҢжүҖд»ҘеҶҷж—Ҙеҝ—зҡ„зәҝзЁӢе’ҢдёҡеҠЎзәҝзЁӢд№Ӣй—ҙе№¶ж— д»Һеұһе…ізі»пјҢдёәдәҶиғҪи®©log4j2еңЁд»»дҪ•зҡ„жғ…еҶөдёӢе°ұиғҪиҺ·еҸ–еҲ°дёҡеҠЎзәҝзЁӢдј е…Ҙзҡ„ж ҮзӯҫеҖјпјҢеңЁејӮжӯҘжғ…еҶөдёӢдҪ йңҖиҰҒеңЁеҗҜеҠЁйЎ№йҮҢй…ҚзҪ®д»ҘдёӢд»Јз Ғпјҡ

public static void main(String[] args) {
    System.setProperty("log4j2.isThreadContextMapInheritable", Boolean.TRUE.toString());
    SpringApplication.run(RunnerApplication.class, args);

еҰӮжһңдҪ дёҚжҳҜSpringbootйЎ№зӣ®пјҢдёҚж–№дҫҝеңЁеҗҜеҠЁйЎ№д»Јз ҒйҮҢиҝӣиЎҢи®ҫзҪ®пјҢиҝҷдёӘеҸӮж•°д№ҹеҸҜд»Ҙи®ҫзҪ®еңЁjvmеҸӮж•°йҮҢпјҡ

            <!--жҺ§еҲ¶еҸ°еҸӘиҫ“еҮәlevelеҸҠд»ҘдёҠзә§еҲ«зҡ„дҝЎжҒҜпјҲonMatchпјүпјҢе…¶д»–зҡ„зӣҙжҺҘжӢ’з»қпјҲonMismatchпјү-->

                <TimeBasedTriggeringPolicy/>

            </Policies>
        </RollingFile>

        <File name="AsyncLogFile" fileName="./logs/log4j2-async.log" append="false" >
            <PatternLayout pattern="%date{HH:mm:ss.SSS}  %-5level [%thread] %logger{36} - %tmsg%n"/>
